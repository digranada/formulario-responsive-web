Actividad de Aprendizaje No. 2

Formulario de Registro con Validación JavaScript

Desarrollo de un formulario de registro responsive con validación básica de campos utilizando JavaScript. Este formulario se adapta a diferentes dispositivos y valida los datos ingresados por el usuario.

Ver Descripción Probar Formulario

Descripción de la Actividad

Objetivo General

Crear un formulario de registro de usuario con diseño responsive que implemente validación básica de campos utilizando JavaScript para garantizar la integridad de los datos ingresados.

Requisitos Específicos

  • Formulario responsive: Diseño que se adapte a diferentes dispositivos (escritorio, tabletas, móviles)
  • Campos básicos: Incluir campos para nombres, apellidos, fecha de nacimiento, país, email y contraseña
  • Validación JavaScript: Implementar validación básica de campos obligatorios
  • Mensajes de error: Mostrar mensajes de error cuando los campos obligatorios no estén completos
  • Botón de registro: Implementar lógica para procesar el formulario cuando todos los campos sean válidos
  • Organización de archivos: Separar HTML, CSS y JavaScript en archivos diferentes

Validaciones Implementadas

Campos obligatorios completos
Formato de email válido
Longitud mínima de contraseña
Fecha de nacimiento válida

Función Principal de Validación


function validarFormularioBasico() {
    // Obtener valores de los campos
    const nombre = document.getElementById('nombreBasico').value.trim();
    const apellido = document.getElementById('apellidoBasico').value.trim();
    const fecha = document.getElementById('fechaBasico').value;
    const pais = document.getElementById('paisBasico').value;
    const email = document.getElementById('emailBasico').value.trim();
    const password = document.getElementById('passwordBasico').value;
    
    // Limpiar errores anteriores
    limpiarErroresBasico();
    
    // Validar cada campo
    let esValido = true;
    
    if (!nombre) {
        mostrarErrorBasico('nombreBasico', 'El nombre es obligatorio');
        esValido = false;
    }
    
    if (!apellido) {
        mostrarErrorBasico('apellidoBasico', 'El apellido es obligatorio');
        esValido = false;
    }
    
    if (!fecha) {
        mostrarErrorBasico('fechaBasico', 'La fecha de nacimiento es obligatoria');
        esValido = false;
    }
    
    if (!pais) {
        mostrarErrorBasico('paisBasico', 'Debe seleccionar un país');
        esValido = false;
    }
    
    if (!email) {
        mostrarErrorBasico('emailBasico', 'El email es obligatorio');
        esValido = false;
    } else if (!validarEmailBasico(email)) {
        mostrarErrorBasico('emailBasico', 'El formato del email no es válido');
        esValido = false;
    }
    
    if (!password) {
        mostrarErrorBasico('passwordBasico', 'La contraseña es obligatoria');
        esValido = false;
    } else if (password.length < 6) {
        mostrarErrorBasico('passwordBasico', 'La contraseña debe tener al menos 6 caracteres');
        esValido = false;
    }
    
    return esValido;
}

Función para Mostrar Mensajes


function mostrarMensajeEstado(mensaje, tipo) {
    const mensajeDiv = document.getElementById('mensajeEstado');
    
    if (mensajeDiv) {
        mensajeDiv.textContent = mensaje;
        mensajeDiv.className = 'mensaje-estado';
        
        if (tipo === 'exito') {
            mensajeDiv.classList.add('mensaje-exito');
        } else if (tipo === 'error') {
            mensajeDiv.classList.add('mensaje-error');
        }
        
        mensajeDiv.style.display = 'block';
        
        // Ocultar mensaje después de 5 segundos
        setTimeout(() => {
            mensajeDiv.style.display = 'none';
        }, 5000);
    }
}

function mostrarErrorBasico(campoId, mensaje) {
    const errorDiv = document.getElementById(`error${capitalize(campoId)}`);
    if (errorDiv) {
        errorDiv.textContent = mensaje;
        errorDiv.style.display = 'block';
        
        // Marcar campo como con error
        const campo = document.getElementById(campoId);
        if (campo) {
            campo.classList.add('campo-error');
        }
    }
}

Event Listeners del Formulario


// Configurar eventos cuando el DOM esté listo
document.addEventListener('DOMContentLoaded', function() {
    // Formulario básico
    const formularioBasico = document.getElementById('formularioRegistroBasico');
    
    if (formularioBasico) {
        formularioBasico.addEventListener('submit', function(event) {
            event.preventDefault();
            
            if (validarFormularioBasico()) {
                // Obtener datos para el mensaje
                const nombre = document.getElementById('nombreBasico').value.trim();
                const apellido = document.getElementById('apellidoBasico').value.trim();
                
                mostrarMensajeEstado(
                    `¡Registro exitoso! Bienvenido/a ${nombre} ${apellido}`,
                    'exito'
                );
                
                // Limpiar formulario después de 3 segundos
                setTimeout(() => {
                    formularioBasico.reset();
                    limpiarErroresBasico();
                }, 3000);
            } else {
                mostrarMensajeEstado(
                    'Por favor, corrija los errores en el formulario',
                    'error'
                );
            }
        });
    }
    
    // Botón de datos de ejemplo
    const btnEjemplo = document.getElementById('btnEjemplo');
    if (btnEjemplo) {
        btnEjemplo.addEventListener('click', function() {
            document.getElementById('nombreBasico').value = 'María';
            document.getElementById('apellidoBasico').value = 'González';
            document.getElementById('fechaBasico').value = '1990-05-15';
            document.getElementById('paisBasico').value = 'EC';
            document.getElementById('emailBasico').value = 'maria@ejemplo.com';
            document.getElementById('passwordBasico').value = 'clave123';
            
            limpiarErroresBasico();
            
            mostrarMensajeEstado(
                'Se han cargado datos de ejemplo. Puede modificarlos si lo desea.',
                'info'
            );
        });
    }
});
